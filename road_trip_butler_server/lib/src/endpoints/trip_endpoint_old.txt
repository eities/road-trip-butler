class TripEndpoint extends Endpoint {
  Future<Trip> createTrip(
    Session session, {
    required String startAddress,
    required String endAddress,
    required DateTime departureTime,
    required String personality,
    required String preferences,
  }) async {
    // 1. Create and Save the Trip record
    var trip = Trip(
      description: 'Trip from $startAddress to $endAddress',
      startAddress: startAddress,
      endAddress: endAddress,
      departureTime: departureTime,
      polyline: '',
      preferences: preferences,
      totalDurationSeconds: 0,
      personality: personality,
    );
    
    // Save to DB to get the trip.id
    trip = await Trip.db.insertRow(session, trip);

    // 2. Setup Dartantic Agent
    final geminiApiKey = session.passwords['geminiApiKey'];
    
    final agent = Agent.forProvider(
      GoogleProvider(apiKey: geminiApiKey!),
      chatModelName: 'gemini-1.5-flash', // Use flash for speed/cost in hackathons
    );

    // 3. Define the Schema for Dartantic
    final stopSchema = {
      "slotTitle": "String",
      "name": "String",
      "address": "String",
      "latitude": "double",
      "longitude": "double",
      "butlerNote": "String",
      "category": "String",
      "priceLevel": ["cheap", "medium", "expensive"], // Gemini should return 'cheap', 'medium', or 'expensive'
    };

    try {
      // 4. Call the structured generation
      // count: 12 ensures we get enough options for the UI
      final List<dynamic> rawStops = await agent.generateList(
        prompt: "Plan a road trip from $startAddress to $endAddress. "
            "The user's personality is: $personality. "
            "Additional preferences: $preferences. "
            "Group stops into logical 'Trip Chapters'.",
        schema: stopSchema,
        count: 12,
      );

      // 5. Convert and Save Stops to Database
      var stops = <Stop>[];
      for (var data in rawStops) {
        var stop = Stop(
          tripId: trip.id!,
          name: data['name'] ?? 'Unknown Location',
          slotTitle: data['slotTitle'] ?? 'Suggested Stop',
          address: data['address'] ?? '',
          // Use .toDouble() to be safe against int vs double errors
          latitude: (data['latitude'] as num).toDouble(),
          longitude: (data['longitude'] as num).toDouble(),
          category: data['category'] ?? 'general',
          butlerNote: data['butlerNote'] ?? '',
          status: StopStatus.untouched,
          // Map string from AI to your Serverpod Enum
          priceLevel: PriceLevel.values.byName(data['priceLevel']?.toString().toLowerCase() ?? 'medium'),
          estimatedArrivalTime: departureTime.add(const Duration(hours: 2)),
        );

        // Save each stop linked to the tripId
        final savedStop = await Stop.db.insertRow(session, stop);
        stops.add(savedStop);
      }

      // 6. Return the trip with stops included
      // We use findById with 'include' to make sure the list is populated for the frontend
      final finalTrip = await Trip.db.findById(
        session,
        trip.id!,
        include: Trip.include(stops: Stop.includeList()),
      );

      return finalTrip!;
    } catch (e) {
      session.log("Butler Error: $e", level: LogLevel.error);
      rethrow; // Or return a Trip with an error note
    }
  }
